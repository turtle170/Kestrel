1//! kestrel-ipc: High-performance IPC between Kestrel Linux guest and Windows host.
2//!
3//! Uses a Windows Named Pipe as the primary channel, carrying:
4//!   - Raw Ethernet frames for networking
5//!   - Serial console data for kestrel-term
6//!   - Framed messages for graphics/block I/O
7//!
8//! On the Linux side (running inside the Kestrel guest), this module
9//! provides a virtual network device that reads/writes to the shared
10//! memory ring buffers exposed by kestrel-bridge.
11
12use anyhow::{Result, Context, bail};

14use log::{info, warn, debug};
15use std::io::{Read, Write};
16use std::sync::{Arc, Mutex};
17use std::thread;
18use std::time::Duration;

20/// Named pipe for Kestrel serial console (kestrel-term).
21pub const KESTREL_SERIAL_PIPE: &str = r"\\.\pipe\KestrelSerial";
22/// Named pipe for Kestrel network packets.
23pub const KESTREL_NET_PIPE: &str = r"\\.\pipe\KestrelNet";

25/// A framed message over the IPC channel.
26/// Frame format: [4 bytes type][4 bytes length][length bytes payload]
27#[repr(u32)]
28#[derive(Debug, Clone, Copy, PartialEq, Eq)]
29pub enum FrameType {
30    Serial  = 0x01,
31    NetTx   = 0x02,
32    NetRx   = 0x03,
33    Graphics = 0x04,
34    Control  = 0xFF,
35}
36
37#[derive(Debug, Clone)]
38pub struct IpcFrame {
39    pub frame_type: u32,
40    pub payload: Vec<u8>,
41}

43/// Windows Named Pipe server for the Kestrel IPC bridge.
44pub struct KestrelIpcServer {
45    serial_pipe_path: String,
46    net_pipe_path: String,
47}
48
49impl KestrelIpcServer {
50    pub fn new() -> Self {
51        Self {
52            serial_pipe_path: KESTREL_SERIAL_PIPE.to_string(),
53            net_pipe_path: KESTREL_NET_PIPE.to_string(),
54        }
55    }
56
57    /// Start the IPC server. Spawns background threads for serial and network pipes.
58    pub fn start(&self) -> Result<IpcHandle> {
59        let (serial_tx, serial_rx): (Sender<Vec<u8>>, Receiver<Vec<u8>>) = bounded(256);
60        let (net_tx, net_rx): (Sender<Vec<u8>>, Receiver<Vec<u8>>) = bounded(256);
61
62        let serial_pipe = self.serial_pipe_path.clone();
63        let net_pipe = self.net_pipe_path.clone();
64        let stx = serial_tx.clone();
65        let ntx = net_tx.clone();
66
67        // Serial pipe server thread
68        thread::spawn(move || {
69            if let Err(e) = run_serial_server(&serial_pipe, stx) {
70                warn!("[IPC] Serial server error: {}", e);
71            }
72        });
73
74        // Network pipe server thread
75        thread::spawn(move || {
76            if let Err(e) = run_net_server(&net_pipe, ntx) {
77                warn!("[IPC] Net server error: {}", e);
78            }
79        });
80
81        info!("[IPC] Kestrel IPC server started");
82        info!("[IPC]   Serial pipe: {}", KESTREL_SERIAL_PIPE);
83        info!("[IPC]   Network pipe: {}", KESTREL_NET_PIPE);
84
85        Ok(IpcHandle { serial_rx, net_rx })
86    }
87}

95/// Run the serial console named pipe server.
96fn run_serial_server(pipe_path: &str, tx: Sender<Vec<u8>>) -> Result<()> {
97    #[cfg(target_os = "windows")]
98    {
99        use windows::Win32::System::Pipes::*;
100        use windows::Win32::Foundation::*;
101        use windows::Win32::Storage::FileSystem::*;
102        use windows::core::PCWSTR;
103
104        info!("[IPC] Creating serial pipe: {}", pipe_path);
105        // Convert path to wide string
106        let wide: Vec<u16> = pipe_path.encode_utf16().chain(std::iter::once(0)).collect();
107
108        let handle = unsafe {
109            CreateNamedPipeW(
110                PCWSTR(wide.as_ptr()),
111                FILE_FLAGS_AND_ATTRIBUTES(0x00000003), // PIPE_ACCESS_DUPLEX
112                PIPE_TYPE_BYTE | PIPE_READMODE_BYTE | PIPE_WAIT,
113                1,      // max instances
114                65536,  // out buffer size
115                65536,  // in buffer size
116                0,      // default timeout
117                None,
118            )
119        };
120
121        if handle == INVALID_HANDLE_VALUE {
122            bail!("CreateNamedPipeW failed for serial pipe");
123        }
124
125        loop {
126            info!("[IPC] Waiting for kestrel-term to connect to serial pipe...");
127            unsafe { ConnectNamedPipe(handle, None) }
128                .context("ConnectNamedPipe failed")?;
129            info!("[IPC] kestrel-term connected to serial console");
130
131            let mut buf = vec![0u8; 1024];
132            loop {
133                let mut bytes_read = 0u32;
134                let ok = unsafe {
135                    ReadFile(
136                        handle,
137                        Some(buf.as_mut_slice()),
138                        Some(&mut bytes_read),
139                        None,
140                    )
141                };
142                if ok.is_err() || bytes_read == 0 {
143                    info!("[IPC] kestrel-term disconnected from serial pipe");
144                    unsafe { DisconnectNamedPipe(handle) }.ok();
145                    break;
146                }
147                let _ = tx.send(buf[..bytes_read as usize].to_vec());
148            }
149        }
150    }
151    #[cfg(not(target_os = "windows"))]
152    {
153        // On Linux: open /dev/ttyS0 for loopback testing
154        use std::fs::OpenOptions;
155        let mut dev = OpenOptions::new().read(true).write(true).open("/dev/ttyS0")
156            .context("Cannot open /dev/ttyS0")?;
157        let mut buf = vec![0u8; 256];
158        loop {
159            match dev.read(&mut buf) {
160                Ok(0) => break,
161                Ok(n) => { let _ = tx.send(buf[..n].to_vec()); }
162                Err(e) => { warn!("[IPC] Serial read error: {}", e); break; }
163            }
164        }
165        Ok(())
166    }
167}

169/// Run the network named pipe server.
170fn run_net_server(pipe_path: &str, tx: Sender<Vec<u8>>) -> Result<()> {
171    info!("[IPC] Network pipe stub (platform-specific implementation pending)");
172    // Placeholder: In production this creates a TAP-like interface
173    // that bridges packets between Linux TCP/IP stack and Windows WFP.
174    // For now, use a TCP loopback socket for cross-platform dev testing.
175    use std::net::TcpListener;
176    let listener = TcpListener::bind("127.0.0.1:9001")
177        .context("Cannot bind net IPC port 9001")?;
178    info!("[IPC] Network IPC listening on 127.0.0.1:9001");
179
180    for stream in listener.incoming() {
181        match stream {
182            Ok(mut s) => {
183                let tx2 = tx.clone();
184                thread::spawn(move || {
185                    let mut buf = vec![0u8; 1500]; // Ethernet MTU
186                    loop {
187                        match s.read(&mut buf) {
188                            Ok(0) => break,
189                            Ok(n) => { let _ = tx2.send(buf[..n].to_vec()); }
190                            Err(_) => break,
191                        }
192                    }
193                });
194            }
195            Err(e) => warn!("[IPC] Net accept error: {}", e),
196        }
197    }
198    Ok(())
199}
